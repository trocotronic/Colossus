modulo modulos/modulo.dll { // para linux, modulos/modulo.so
	nick nickdelbot;
	ident identdelbot;
	host hostdelbot;
	realname "Realname del bot";
	modos modosdelbot;
	config modulo.inc;  //opcional
	residente "#canal1,#canal2...";  // opcional
};
